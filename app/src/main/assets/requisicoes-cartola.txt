 func webViewDidFinishLoad(webView: UIWebView) {
	let storage = NSHTTPCookieStorage.sharedHTTPCookieStorage()
        for cookie in storage.cookies! {
            if cookie.name.containsString("GLBID"){
                print(cookie.value)
                loading()
                
                NSUserDefaults.standardUserDefaults().setValue(cookie.value, forKey: "TOKEN")
                
                self.meuTimeCartola()
                self.AdicionaLigaRealm()
                
                NSNotificationCenter.defaultCenter().postNotificationName("AtualizaMeuTime", object: nil, userInfo: nil)
                
                if self.navigation {
                    NSNotificationCenter.defaultCenter().postNotificationName("CanceladoLogin", object: nil)
                    self.removeLoading()
                }else{
                    self.dismissViewControllerAnimated(false, completion: { 
                        self.removeLoading()
                    })
                }
                
                webView.stopLoading()
            }
        }
}


* Autenticaçaõ: Necessário para pegar token para chamadas privadas
https://login.globo.com/api/authentication.

- Autenticação para obter o token

Tem que fazer um post via json nessa url:

https://login.globo.com/api/authentication

- O conteúdo do json tem que seguir esse padrão:

{"payload":{"email":"[EMAIL]","password":"[SENHA]","serviceId": 438}}

		** Eu particulamente salvo o token e uso ele em quase todos requests, esse token não expira, pelo menos até hoje não expirou rsrs. Se expirar só fazer outro login pra gerar outro token.


- O retorno do post vai trazer esse json:

{"id":"Authenticated","userMessage":"Usuário autenticado com sucesso","glbId":"[TOKEN]"}
